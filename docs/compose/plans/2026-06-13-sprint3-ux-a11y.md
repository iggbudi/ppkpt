### Task 1: Step Form Validation

**Covers:** [S3-1]

**Files:**
- Modify: `public/app.js`

- [ ] **Step 1: Add validation function to app.js**

In `setupEventListeners()`, add step validation:

```javascript
function validateStep(step) {
  var section = document.getElementById('formStep' + step);
  if (!section) return true;
  
  var requiredFields = section.querySelectorAll('[required]');
  var firstInvalid = null;
  
  requiredFields.forEach(function(field) {
    if (!field.value || (field.value.trim && field.value.trim() === '')) {
      field.style.borderColor = 'var(--danger)';
      if (!firstInvalid) firstInvalid = field;
    } else {
      field.style.borderColor = '';
    }
  });
  
  if (firstInvalid) {
    firstInvalid.focus();
    showTopSystemAlert('Harap isi semua field yang wajib diisi.');
    return false;
  }
  
  return true;
}
```

- [ ] **Step 2: Add validation to step navigation**

  var modal = document.getElementById(modalId);

}

}
